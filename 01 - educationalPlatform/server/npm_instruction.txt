//В нашем случае index.js
//Пишем в терминале, при переключении на корень index.js
cd server
npm init -y //иницилизирует проект, после должен появиться package.json (описание проекта и зависимости)

//Далее подключаем фреймворки, бд (PostgreSQL), sequlize (ORM), и чтобы мы могли обращаться с браузера к серверу - нужно установить ядро (cors), ичтобы задавать переменные окржуения (dotenv)
npm install express pg pg-hstore sequelize cors dotenv
//после установки все модули можно увидеться в dependencies(package.json)
//так же как завимисоть для dev установим nodemon
